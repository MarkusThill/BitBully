176 def play(self, move: int | Sequence[int] | str) -> bool:
177 """Plays one or more moves for the current player.
178
179 The method updates the internal board state by dropping tokens
180 into the specified columns. Input can be:
181 - a single integer (column index 0 to 6),
182 - an iterable sequence of integers (e.g., `[3, 1, 3]` or `range(7)`),
183 - or a string of digits (e.g., `"33333111"`) representing the move order.
184
185 Args:
186 move (int | Sequence[int] | str):
187 The column index or sequence of column indices where tokens should be placed.
188
189 Returns:
190 bool: True if the move was played successfully, False if the move was illegal.
191
192
193 Example:
194 Play a sequence of moves into the center column (column index 3):
195 ```python
196 import bitbully as bb
197
198 board = bb.Board()
199 assert board.play([3, 3, 3]) # returns True on successful move
200 board
201 ```
202
203 Expected output:
204
205 ```
206 _ _ _ _ _ _ _
207 _ _ _ _ _ _ _
208 _ _ _ _ _ _ _
209 _ _ _ X _ _ _
210 _ _ _ O _ _ _
211 _ _ _ X _ _ _
212 ```
213
214 Example:
215 Play a sequence of moves across all columns:
216 ```python
217 import bitbully as bb
218
219 board = bb.Board()
220 assert board.play(range(7)) # returns True on successful move
221 board
222 ```
223 Expected output:
224 ```text
225 _ _ _ _ _ _ _
226 _ _ _ _ _ _ _
227 _ _ _ _ _ _ _
228 _ _ _ _ _ _ _
229 _ _ _ _ _ _ _
230 X O X O X O X
231 ```
232
233 Example:
234 Play a sequence using a string:
235 ```python
236 import bitbully as bb
237
238 board = bb.Board()
239 assert board.play("33333111") # returns True on successful move
240 board
241 ```
242 Expected output:
243 ```text
244 _ _ _ _ _ _ _
245 _ _ _ X _ _ _
246 _ _ _ O _ _ _
247 _ O _ X _ _ _
248 _ X _ O _ _ _
249 _ O _ X _ _ _
250 ```
251 """
252 # Case 1: string -> pass through directly
253 if isinstance(move, str):
254 return self._board.play(move)
255
256 # Case 2: int -> pass through directly
257 if isinstance(move, int):
258 return self._board.play(move)
259
260 # From here on, move is a Sequence[...] (but not str or int).
261 move_list: list[int] = [int(v) for v in cast(Sequence[Any], move)]
262 return self._board.play(move_list)

278 def set_board(self, board: Sequence[int] | Sequence[Sequence[int]] | str) -> bool:
279 """Sets (overrides) the board to a specific state.
280
281 Args:
282 board (Sequence[int] | Sequence[Sequence[int]] | str):
283 The new board state. Accepts:
284 - 2D array (list, tuple, numpy-array) with shape 7x6 or 6x7
285 - 1D sequence of ints: a move sequence of columns (e.g., [0, 0, 2, 2, 3, 3])

288 Returns:
289 bool: True if the board was set successfully, False otherwise.

291 # TODO: also allow other types for `board`, e.g., numpy arrays and convert to a list of lists
292 if isinstance(board, str):
293 return self._board.setBoard(board)

295 # From here on, board is a Sequence[...] (but not str).
296 # Distinguish 2D vs 1D by inspecting the first element.

298 # Case 2: 2D -> list[list[int]]

301 return self._board.setBoard(grid)
302
303 # Case 3: 1D -> list[int]
304 moves: list[int] = [int(v) for v in cast(Sequence[Any], board)]
305 return self._board.setBoard(moves)
